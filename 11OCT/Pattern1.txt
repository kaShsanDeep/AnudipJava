package LabProjects;

public class Pattern1 {

	public static void main(String[] args) {
		  
		
		 for(int i=0;i<=4;i++) {
			
					 
			 for (int j=0;j<=4;j++) {
			 if(i==0 || i==4 || j==3  || j==4) { 
				 System.out.print("*");			 
			 }			 
			 System.out.print(" ");
			
			 	}
			 System.out.println();
			 }
			 
			
		 }
		 
	}
